dva-core-ts: 2.0.7 // dva 状态管理
@react-navigation/stack: 6.3.10 // 路由
@react-navigation/bottom-tabs: 6.5.2 // 底部导航
@react-navigation/material-top-tabs: 6.5.1 // 顶部导航
